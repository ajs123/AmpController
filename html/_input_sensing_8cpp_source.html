30void InputSensing::signalChecks() {
31
32 // Triggers
33 digitalTriggerPresent = !digitalRead(PIN_DIGITAL_TRIGGER);
34 analogTriggerPresent = !digitalRead(PIN_ANALOG_TRIGGER);
35
36 // Analog levels
37 // The simplest approach is that anything over threshold counts
38 // This is not robust against little spikes, etc.
39
40 bool analogDetected = false;
41
42 uint16_t aSignal = max(analogRead(PIN_ANALOG_L), analogRead(PIN_ANALOG_R));
43 if (aSignal >= ANALOG_SIGNAL_THRESHOLD) {
44 lastSignalTime = millis();
45 analogDetected = true;
46 analogSignalPresent = true;
47 } else {
48 analogSignalPresent = !( (millis() - lastSignalTime) > ANALOG_SIGNAL_HOLDUP );
49 }
50
51 if (analogDetected) {
52 analogImpedanceLow = true;
53 } else {
54 analogImpedanceLow = analogImpedanceTest();

59bool InputSensing::analogImpedanceTest() {
60
61 // If an input is disconnected, or connected to a high-impedance (turned off)
62 // source, then turning on the input pullup should bring the pin voltage very close
63 // to Vdd. Otherwise, it must be connected to a source.
64 //
65 // This will read incorrectly if the signal came on right before this was called,
66 // but that will get corrected on the next update.
67
68 uint16_t baseline;
69 uint16_t purturbed;
70 constexpr uint16_t thresh = (uint16_t) (4096 * 3.3 / 3.6 * ANALOG_IMPEDANCE_THRESHOLD);
71
72 pinMode(PIN_ANALOG_L, INPUT_PULLUP);
73 pinMode(PIN_ANALOG_R, INPUT_PULLUP);
74 delay(10);
75 bool high = (analogRead(PIN_ANALOG_L) > thresh) && (analogRead(PIN_ANALOG_R) > thresh);
76 pinMode(PIN_ANALOG_L, INPUT);
77 pinMode(PIN_ANALOG_R, INPUT);
