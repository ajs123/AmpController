6bool Remote::handleRepeats() {
7 // NEC repeat flag (provided by the IRRemote library)
8 if ((protocolNum == NEC) &&
9 (lastProtocol == NEC) &&
10 (value == REPEAT_CODE) &&
11 (receivedTime - lastReceiveTime <= maxDelayBeforeRepeat)) {
12 value = lastCommand;
13 return true;
14 }
15 return false;
16}
