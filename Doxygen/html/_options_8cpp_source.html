25int8_t Options::readParamFile(const char * name, void * value, const uint8_t length) {
26 char path[MAX_PATH_LENGTH];
27 makePath(path, name, MAX_PATH_LENGTH);
28
29 Serial.printf("\nChecking for %s...\n", path);
30 if (!InternalFS.exists(path)) return 1;
31 if (!file.open(path, FILE_O_READ)) return -1;
32
33 uint8_t trimmedLength = min(length, MAX_VARIABLE_SIZE);
34 uint8_t nRead = file.read(value, trimmedLength);
35 file.close();
36 Serial.printf("Read %d bytes (expected %d).\n", nRead, trimmedLength);
37 if (nRead != trimmedLength) return -2;
38
39 return 0;
40}

42int8_t Options::writeParamFile(const char * name, const void * value, uint8_t length) {
43 char path[MAX_PATH_LENGTH];
44 makePath(path, name, MAX_PATH_LENGTH);
45
46 Serial.printf("\nWriting %s...\n", path);
47 if (InternalFS.exists(path)) InternalFS.remove(path); // Remove and replace preferred over open and seek (?)
48 if (!file.open(path, FILE_O_WRITE)) return -1;
49
50 uint8_t trimmedLength = min(length, MAX_VARIABLE_SIZE);
51 uint8_t nWritten = file.write((const char *) value, trimmedLength);
52 file.close();
53 Serial.printf("Wrote %d bytes (expected %d)\n", nWritten, trimmedLength);
54 if (nWritten != trimmedLength) return -2;
55

75bool Options::save() {
76 char buf[MAX_VARIABLE_SIZE];
77 bool error = false;
78
79 for (uint8_t i = 0; i < sizeof(optionTable)/sizeof(writableOption_t); i++) {
80
81 uint8_t trimmedLength = min(optionTable[i].size, MAX_VARIABLE_SIZE);
82
83 // Check what's already in flash for this variable.
84 int8_t result = readParamFile(optionTable[i].name, &buf, trimmedLength);
85 if (result < 0) {
86 error = true;
87 continue;
88 }
89 else if (result == 0) {
90 Serial.printf("\nsave() of %s found \n", optionTable[i].name);
91 Serial.print("Flash: ");
92 printHex(buf, optionTable[i].size);
93 Serial.print("Current: ");
94 char bufCurrent[MAX_VARIABLE_SIZE];
95 memcpy(bufCurrent, optionTable[i].value, optionTable[i].size);
96 printHex(bufCurrent, optionTable[i].size);
97 // If what's written there is the same as the current value, nothing to write.
98 if (!memcmp(buf, optionTable[i].value, optionTable[i].size)) continue;
99 }
100
101 result = writeParamFile(optionTable[i].name, optionTable[i].value, optionTable[i].size);
102 error |= (result < 0);
103 }
104
105 return error;
106}
