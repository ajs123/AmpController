43// The maximum volume can't be set above the compile-time maximum.
44// And the volume at startup can't be higher than the absolute maximum.
45// If the absolute maximum is lowered and then raised again without
46// having edited the volume at startup, track appropriately.
47
48result handleMaxVolume(eventMask event) {
49 f_maxVolume = min(f_maxVolume, MAXIMUM_VOLUME / -0.5);
50 f_maxInitialVolume = min(saved_f_maxInitialVolume, f_maxVolume);
51 return proceed;
52}

111void padString(char * string, const uint8_t length) {
112 uint8_t currentLength = strlen(string);
113 if (currentLength >= MAX_LABEL_LENGTH) return;
114 uint8_t len = min(length, MAX_LABEL_LENGTH);
115 memset(string + currentLength, 32, MAX_LABEL_LENGTH - currentLength);
116}
117
118void fixString(char * string, const char * defaultLabel) {
119 if (string[0] == 32) {
120 strncpy(string, defaultLabel, MAX_LABEL_LENGTH);
121 return;
122 }
123 uint8_t currentLength = strlen(string);
124 for (uint8_t i = currentLength - 1; i > 0; i--) {
125 if (string[i] != 32) break;
126 string[i] = 0;
127 }
128}
