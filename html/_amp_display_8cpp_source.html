265 void AmpDisplay::displayLRBarGraph(uint8_t left, uint8_t right, areaSpec_t area)
266 {
267 // Erase the area
268 display->setDrawColor(0);
269 display->drawBox(area.XL, area.YT, area.XR - area.XL, area.YB - area.YT);
270
271 // The bar graphs extend left and right from the center of the area
272 // There's a two-pixel gap in the center
273 // The area is assumed to have an even width
274 uint8_t barWidth = (area.XR - area.XL + 1) / 2 ; // Assumes total area width is an even number
275 uint8_t leftAreaXR = area.XL + barWidth - 2;
276 uint8_t rightAreaXL = leftAreaXR + 2;
277
278 uint8_t leftWidth = max( ( (int) left * (int) (barWidth) ) / 100, 1);
279 uint8_t rightWidth = max( ( (int) right * (int) (barWidth) ) / 100, 1);
280
281 display->setDrawColor(1);
282 display->drawBox(leftAreaXR - leftWidth, area.YT + 2, leftWidth, area.YB - area.YT - 2);
283 display->drawBox(rightAreaXL, area.YT + 2, rightWidth, area.YB - area.YT - 2);
284 display->updateDisplay(); // Re-draw only, without full refresh and wakeup
285 }
